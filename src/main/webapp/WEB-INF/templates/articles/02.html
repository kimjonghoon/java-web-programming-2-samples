Thymeleaf extras springsecurity6 모듈은, 타임리프 템플릿에 시큐리티 관련 작업을 핸들링할 수 있는 기능을 제공함으로써, 타임리프와 스프링 시큐리티를 통합한다.

이 모듈은 새로운 다이얼렉Dialect, org.thymeleaf.extras.springsecurity6.dialect.SpringSecurityDialect을 제공한다. 이 다이얼렉은 디폴트 접두사는 sec다.

표현식 유틸리티 객체:
#authentication: 스프링 시큐리티 인증 객체를 나타낸다.
#authorization: 표현식에 기반한 권한을 검사하기 위한 메소드를 제공한다.

속성:
sec:authentication="prop": 인증 객체의 속성을 출력한다.
sec:authorize="expr" 또는 sec:authorize-expr="expr": 지정된 표현식에 권한을 가진 사용자라면 자식 엘리먼트를 랜더링한다.
sec:authorize-url="url": 인증된 사용자에게 지정된 URL를 볼 수 있는 권한을 있다면 자식 엘리먼트를 랜더링한다.
sec:authorize-acl="object :: permissions": 인증된 사용자에게 지정된 도메인 객체에 대한 지정된 권한이 있는 경우 자식 엘리먼트를 랜더링한다.

자바 기반 설정

요청을 처리하는 프로세스에 스프링 시큐리티가 참여하는 지를 먼저 확인하는 게 좋다.
다음 스프링 시큐리티 자바 기반 설정 클래스는 그런 목적으로 작성했다.

SecurityConfig.java
package net.java_school.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.core.userdetails.User;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.security.core.userdetails.UserDetailsService;
import org.springframework.security.provisioning.InMemoryUserDetailsManager;
import org.springframework.security.web.SecurityFilterChain;
import static org.springframework.security.config.Customizer.withDefaults;

@Configuration
@EnableWebSecurity
public class SecurityConfig {
  @Bean
  public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
    http.authorizeHttpRequests((authorize) -> authorize.anyRequest().authenticated())
        .httpBasic(withDefaults())
        .formLogin(withDefaults());
    return http.build();
  }
  @Bean
  public UserDetailsService userDetailsService() {
    UserDetails user = User.withDefaultPasswordEncoder()
        .username("user")
        .password("password")
        .roles("USER")
        .build();
    return new InMemoryUserDetailsManager(user);
  }
}

다음 스프링 시큐리티 자바 기반 설정 클래스가 없으면 스프링 시큐리티가 동작하지 않는다.
AbstractSecurityWebApplicationInitializer를 상혹하는 클래스는 DelegatingFilterProxy를 등록하여 다른 필터보다 먼저 springSecurityFilterChain를 사용한다.
ContextLoaderListener도 등록한다.

}

이것만으로도 스프링 시큐리티가 동작하지만, HttpSessionEventPublisher를 리슨너에 포함시키려면 enableHttpSessionEventPublisher() 메소드를 오버라이딩한다.
세션 매니지먼트가 지정된 최대 세션 수를 가지게 하려면 true를 리턴하게 한다.

HttpSessionEventPublisher는 스프링 루트 WebApplicationContext에 HttpSessionApplicationEvents를 게시한다.

SecurityWebApplicationInitializer.java
package net.java_school.config;

import org.springframework.security.web.context.AbstractSecurityWebApplicationInitializer;

public class SecurityWebApplicationInitializer extends AbstractSecurityWebApplicationInitializer {
  @Override
  protected boolean enableHttpSessionEventPublisher() {
    return true;
  }
}

스프링 시큐리티 자바 기반 설정 클래스를 아래처럼 수정한다.

SecurityConfig.java
package net.java_school.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.http.HttpMethod;
import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.SecurityFilterChain;

import static org.springframework.security.config.Customizer.withDefaults;

import javax.sql.DataSource;

import org.springframework.beans.factory.annotation.Autowired;


@Configuration
@EnableWebSecurity
public class SecurityConfig {
  @Autowired
  private DataSource dataSource;

  @Bean
  public PasswordEncoder passwordEncoder() {
    return new BCryptPasswordEncoder();
  }

  @Autowired
  public void configureGlobal(AuthenticationManagerBuilder builder) throws Exception {
    builder.jdbcAuthentication().dataSource(dataSource).passwordEncoder(this.passwordEncoder());
  }

  @Bean
  public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
    http
      .authorizeHttpRequests((authorize) -> authorize
          .requestMatchers(HttpMethod.GET, "/static/**").permitAll()
          .requestMatchers(HttpMethod.GET, "/users/signUp").permitAll()
          .requestMatchers(HttpMethod.POST,"/users/signUp").permitAll()
          .requestMatchers(HttpMethod.GET, "/users/welcome").permitAll()
          .anyRequest().authenticated()
      )
      .formLogin(form -> form.loginPage("/users/login").permitAll().loginProcessingUrl("/login"))
      .httpBasic(withDefaults());
    return http.build();
  }
}

User.java
package net.java_school.user;

import java.util.List;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Size;

public class User {
  @NotNull
  @Size(min=4, message="{username.validation.error}")
  private String username;

  @NotNull
  @Size(min=4, message="{password.validation.error}")
  private String password;
	
  private Integer enabled;

  private List<String> authorities;

  public User() {}
  
  public User(String username, String password, Integer enabled, List authorities) {
    this.username = username;
    this.password = password;
    this.enabled = enabled;
    this.authorities = authorities;
  }
  public String getUsername() {
    return username;
  }
  public void setUsername(String username) {
    this.username = username;
  }
  public String getPassword() {
    return password;
  }
  public void setPassword(String password) {
    this.password = password;
  }
  public Integer getEnabled() {
    return enabled;
  }
  public void setEnabled(Integer enabled) {
    this.enabled = enabled;
  }
  public List<String> getAuthorities() {
    return authorities;
  }
  public void setAuthorities(List<String> authorities) {
    this.authorities = authorities;
  }
}

  .logout((logout) -> logout.logoutSuccessUrl("/"))

return http.build();

스프링 시큐리티의 패스워드 인코더는 솔트 값을 사용하여 인코드하기에, 인코드되기 전에 같은 값이라도 인코드된 후는 값이 같지 않게 된다.

UserServiceImpl.java
@Override
public int changePassword(String currentPassword, String newPassword, String username) {
  String dbPassword = this.getPassword(username);
  boolean check = this.bcryptPasswordEncoder.matches(currentPassword, dbPassword);
  if (check == false) throw new AccessDeniedException("The password is incorrect!");
  newPassword = this.bcryptPasswordEncoder.encode(newPassword);
  return userMapper.updatePassword(newPassword, username);
}
@Override
public String getPassword(String username) {
  return userMapper.selectPassword(username);
}

메소드 매개변수를 이용하는 메소드 보안

참조:
https://docs.spring.io/spring-security/reference/servlet/authorization/method-security.html#using_method_parameters

스프링 시큐리티는 매서드 매개변수를 찾는 방법을 제공하기에 SpEL 표현식에서 매개변수에 접근할 수 있다.
UserService 인터페이스에서 비밀번호를 찾는 메서드는 ADMIN 권한을 가지거나 해당 사용자만 실행할 수 있게 하려면 다음과 같다.

@PreAuthorize("hasRole('ADMIN') or #username == principal.username")
public String getPassword(@P("username") String username);

@P 애너테이션을 사용하기 위해선, 다음 임포트 문이 필요하다.

import org.springframework.security.access.method.P;

매개변수가 하나 이상인 경우는 스프링 Data의 @Param 애너테이션을 사용한다.

이경우 스프링 Data를 의존성에 추가해야 한다.
우리의 예제는 스프링 Data를 사용하지 않으며, getPassword 메서드도 매개변수가 하나이므로 @Param 애너테이션을 사용하지 않아도 된다.

@Param 애너테이션을 테스트하려면 다음 과정을 따른다.

<!-- https://mvnrepository.com/artifact/org.springframework.security/spring-security-data -->
<dependency>
  <groupId>org.springframework.security</groupId>
  <artifactId>spring-security-data</artifactId>
  <version>6.3.4</version>
</dependency>

스프링 시큐리티 설정 클래스에 다음을 추가한다.

@Bean
public SecurityEvaluationContextExtension securityEvaluationContextExtension() {
  return new SecurityEvaluationContextExtension();
}

다음 임포트 문이 필요하다.
import org.springframework.security.data.repository.query.SecurityEvaluationContextExtension;

UserService 인터페이스의 getPassword 메소드에 @Param 애너테이션을 적용한다.

@PreAuthorize("hasRole('ADMIN') or #username == principal.username")
public String getPassword(@Param("username") String username);

다음 임포트 문이 필요하다.
import org.springframework.data.repository.query.Param;

우리의 예제는 Spring Data Integration를 사용하지 않겠다.
Spring Security provides Spring Data integration that allows referring to the current user within your queries. It is not only useful but necessary to include the user in the queries to support paged results since filtering the results afterwards would not scale. https://docs.spring.io/spring-security/reference/features/integrations/data.html
